```cpp

#define SOCKET_SERVER_ADDRESS ""
#define SOCKET_SERVER_PORT 3000

#define SMART_RING_HOST_NAME ""
#define SMART_RING_OTA_PASSWORD ""

#define MICROPHONE_THRESHOLD 580
```
